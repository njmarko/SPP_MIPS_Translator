469  // evaluate any further and no moving from memory to immediate will happen
470  Variables a_regs;
471  for each (Instruction* i_var in instr)
472  {
473  if (i_var->getDst().size() > 0 && find(a_regs.cbegin(), a_regs.cend(), i_var->getDst().front()) != a_regs.cend())
474  {
475  // if the register that holds the adress is being redefined, remove it from the list of address registers
476  a_regs.remove(i_var->getDst().front());
477  }
478  if (i_var->getType() == InstructionType::I_LA)
479  {
480  // keep track of all the registers that hold memory addresses
481  a_regs.push_back(i_var->getDst().front());
482  }
483 
484  if (i_var->getType() == InstructionType::I_SW) {
490  if (i_var->getNumValue() != 0)
491  {
492  return;
493  }
502  if (find(a_regs.cbegin(), a_regs.cend(),i_var->getSrc().back()) == a_regs.cend())
503  {
504  return;
505  }
506  }
507  }
508

510  Variables m_vars_candidates;
511

513  for each (Variable* m_var in m_vars)
514  {
519  if (m_var->getValue() < -32768 || m_var->getValue() > 32767)
520  {
521  continue; // if it is not in range then move to the next mem var
522  }
523

531  m_vars_candidates.push_back(m_var); // add it right away, and only remove it if sw writes to that location

535  {
536  // if the register that holds the adress is being redefined, remove it from the list of address registers
537  address_regs.remove(i_var->getDst().front());
538  }
539 
540  if (i_var->getType() == InstructionType::I_LA)

544  {
545  // this is a potential candidate for replacement by immediate operator if it turns out the memory is read only
546  address_regs.push_back(i_var->getDst().front());
547  }
548  }
549  if (i_var->getType() == InstructionType::I_SW) {
550  // if sw instruction writes to the memory location then that memory location cant be transformed into immediate operand
551  if (find(address_regs.cbegin(),address_regs.cend(),i_var->getSrc().back()) != address_regs.cend())

625  i_var->setNumValue(m_var->getValue()); // assigns the mem value as immediate operand
626  i_var->setInstrType(InstructionType::I_ADDI); // change to type of the instruction to addi
627  }
628  }
629 
630  // this is at the end because instruction can use the reg with value from memory and at the same time define it (example add r1,r1,r2)
631  if (i_var->getDst().size() > 0 && find(regs_loaded_vals.cbegin(), regs_loaded_vals.cend(), i_var->getDst().front()) != regs_loaded_vals.cend())
632  {
633  // if the register that holds value is being redefined, remove it from the list of registers with loaded values
634  regs_loaded_vals.remove(i_var->getDst().front());
635  }
636 
637  }

715 {

719

733 SimplificationStack & ResourceAllocation::performSimplification(InterferenceGraph & ig, SimplificationStack & sims, Variables & vars)
734 {
735  // TODO: Change map key from variable pointer to string so the order is always the same for the same input code
736 
740  std::map <Variable*, int> varRang = makeVarInterferenceMap(ig,vars);
741 
742  // Repeat untill the stack is filled or runtime_error is thrown
743  while (sims.size() < vars.size()) {
744 
745  // First search for the reg variable with the highest rank that is lower than the number of available registers
746  Variable* highestRank = nullptr;
747  for each (std::pair<Variable*,int> p in varRang)
748  {
749  if (p.second < __REG_NUMBER__)
750  {
751  if (highestRank == nullptr) {
752  highestRank = p.first;
753  }
754  else if (p.second > varRang[highestRank])
755  {
756  highestRank = p.first;
757  }
758  }
759 
760  }
761  if (highestRank == nullptr) // if there is no suitable register that means there is a spill that needs to be adressed
762  {
763  throw std::runtime_error("Spill happened while performing the simplification procedure!");
764  }
765  // add the variable to the simplification stack
766  sims.push(highestRank);
767  // reduce the rank of all the neighbours of the variable
768  for each (std::pair<Variable*,int> p in varRang)
769  {
770  if (ig.getIGMatrix()[p.first->getPos()][highestRank->getPos()] == __INTERFERENCE__) {
771  --varRang[p.first];
772  }
773  }
774  // remove the variable from the map
775  varRang.erase(highestRank);
776  }
777  return sims;
778 }

780 void ResourceAllocation::selectRegisters(InterferenceGraph & ig, SimplificationStack & ss)
781 {
782  // set is used because we don't need duplicate register names
783  std::set<Regs> allocated_regs;
784 
785  // we need to go through the enitre simplification stacks
786  while (!ss.empty()) {
787  Variable* v = ss.top();
788  ss.pop();
789  int color = -1;
790  allocated_regs.clear(); // each register has different interferences, so this set gets cleared
791  for (Variables::const_iterator cit = ig.getVars()->cbegin(); cit != ig.getVars()->cend(); ++cit) {
792  int pos = (*cit)->getPos();
793  if (ig.getIGMatrix()[v->getPos()][pos] == __INTERFERENCE__ && (*cit)->getAsignment() != Regs(0)) {
794  // if the interference exists between the two variables, and it has an assigned register
795  allocated_regs.emplace((*cit)->getAsignment());
796  }
797  }
798  // look for the lowest numbered register color that is not taken by the neighbours of the variable
799  bool taken = false;
800  for (int i = 0; i < __REG_NUMBER__; i++)
801  {
802  taken = false;
803  for each (Regs var in allocated_regs)
804  {
805  if (Regs(i+1) == var) {
806  taken = true;
807  break;
808  }
809  }
810  if (!taken) {
811  color = i+1;
812  break;
813  }
814  }
815  if (color == -1) {
816  // if non of the standard available registers can be assigned
817  throw std::runtime_error("Spill happened while performing register allocations!");
818  }
819  else
820  {
821  v->set_assignment(Regs(color));
822  }
823  }
824 }
